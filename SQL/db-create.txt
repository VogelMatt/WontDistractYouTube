

CREATE TABLE UserProfile (
  Id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
  Name VARCHAR(55) NOT NULL,
  Email VARCHAR(255) NOT NULL,
  DisplayName VARCHAR(255) NOT NULL,
  FirebaseUserId VARCHAR(28)
);

CREATE TABLE Video (
  Id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
  Url VARCHAR(255) NOT NULL,
  Title VARCHAR(55) NOT NULL,
  Info VARCHAR(255) NOT NULL,
TopicId INT NOT NULL,
UserProfileId INT NOT NULL,
  CONSTRAINT FK_Video_Topic FOREIGN KEY (TopicId) REFERENCES Topic(Id) ON DELETE CASCADE,
  CONSTRAINT FK_Video_UserProfile FOREIGN KEY (UserProfileId) REFERENCES UserProfile(Id) ON DELETE CASCADE
);

CREATE TABLE VideoTag (
  Id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
  TagId INT NOT NULL,
  VideoId INT NOT NULL,
  CONSTRAINT FK_VideoTag_Tag FOREIGN KEY (TagId) REFERENCES Tag(Id) ON DELETE CASCADE,
  CONSTRAINT FK_VideoTag_Video FOREIGN KEY (VideoId) REFERENCES Video(Id) ON DELETE CASCADE
);

CREATE TABLE Tag (
  Id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
  Name VARCHAR(55) NOT NULL
);

CREATE TABLE Topic (
  Id INT PRIMARY KEY NOT NULL IDENTITY(1,1),
  Title VARCHAR(55) NOT NULL
);
